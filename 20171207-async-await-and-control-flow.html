Control flow when using async & await

Agenda

  • Concurrency
  • Inside callback

await suspends the current async function's evaluation, including all control structures. 

async function httpGet(url) {
  const content = await fetch(url);

  console.log(content);
}

Concurrency

  1. Sequential
  2. Concurrent

Example 1

async function getData() {
  const res1 = await fetch('http://my.api/1');
  const res2 = await fetch('http://my.api/2');

  // do work
}

async function getData() {
  const [res1, res2] = await Promise.all(
    fetch('http://my.api/1'),
    fetch('http://my.api/2')
  );

  // do work
}

Example 2

async function doSteps() {
  const [res1, res2] = await Promise.all(
    httpPost('http://my.api/step1'),
    httpPost('http://my.api/step2')
  );

  // do work
}

async function doSteps() {
  const res1 = httpPost('http://my.api/step1');
  const res2 = httpPost('http://my.api/step2');

  // do something
}

Example 3

async function attack(url) {
  while(true) {
    await fetch(url)

Inside a Callback

  1. forEach
  2. map
async function showData(urls) {
  urls.forEach(url => {
    const data = await fetch(url);
    console.log(data);
  })
}

async function showData(urls) {
  urls.forEach(async (url) => {
    const data = await fetch(url);
    console.log(data);
  })
}

forEach is synchronous but callback is asynchronous

async function showData(urls) {
  for (url of urls) {
    const data = await fetch(url);
    console.log(data);
  }
}

async function showData(urls) {
  for (let i = 0; i < urls.length; i++) {
    const url = urls[i];
    const data = await fetch(url);
    console.log(data);
  }
}

Concurrent version

async function showData(urls) {
  for (url of urls) {
    const data = await fetch(url);
    console.log(data);
  }
}

async function showData(urls) {
  const promises = urls.map(url => {
    return fetch(url);
  });

  await Promise.all(promises);
}

async function showData(urls) {
  const promises = urls.map(url => {
    return fetch(url);
  });

  return Promise.all(promises);
}

function showData(urls) {
  const promises = urls.map(url => {
    return fetch(url);
  });

  return Promise.all(promises);
}
